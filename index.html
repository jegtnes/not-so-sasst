Extend all the things

Extends

						
%geronimo {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}

.massive-header {
  @extend %geronimo;
}

.allons-y {
  @extend %geronimo;
}

.dalek {
  @extend %geronimo;
}

…compiles into

						
.massive-header,
.allons-y,
.dalek {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}

Mixins

						
@mixin geronimo() {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}

.massive-header {
  @include geronimo();
}

.allons-y {
  @include geronimo();
}

.dalek {
  @include geronimo();
}

…compiles into

						
.massive-header {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}

.allons-y {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}

.dalek {
  font-size: mahoosive;
  font-weight: stonking;
  color: unicornHorn;
}
